15std::shared_ptr<BvhTreeNode> Bvh::RecursiveBuild(std::vector<EntityInfo>& entityInfo, int start, int end, int& nodeNumber, std::vector<std::shared_ptr<Entity>>& orderedEntites) {
16 nodeNumber++;
17 std::shared_ptr<BvhTreeNode> root = std::make_shared<BvhTreeNode>();
18 BoundingBox3f totalBounds;
19 for (int i = start; i < end; i++) totalBounds = BoundingBoxUnion(totalBounds, entityInfo[i].bounds);
20 root->bounds = totalBounds;
21 int nEntities = end - start;
22 if (nEntities == 1) {
23 //leaf node
24 root->entityOffset = orderedEntites.size();
25 root->nEntites = 1;
26 orderedEntites.push_back(entites[entityInfo[start].EntityId]);
27 }
28 else {
29 BoundingBox3f centerBounds;
30 for (int i = start; i < end; i++) centerBounds = BoundingBoxUnion(centerBounds, BoundingBox3f(entityInfo[i].center));
31 int dim = -1;
32 double maxD = 0;
33 for (int i = 0; i < 3; i++) {
34 double D = centerBounds.pMax[i] - centerBounds.pMin[i];
35 if (maxD < D) maxD = D, dim = i;
36 }
37 if (dim == -1) {
38 //leaf node
39 root->entityOffset = orderedEntites.size();
40 root->nEntites = nEntities;
41 for (int i = start; i < end; i++) {
42 orderedEntites.push_back(entites[entityInfo[i].EntityId]);
43 }
44 }
45 else {
46 //interior node
47 root->splitAxis = dim;
48 int mid = (start + end) >> 1, success = 1;
49
50 if (splitMethod == SplitMethod::Middle) {
51 double pmid = 0.5 * (centerBounds.pMin[dim] + centerBounds.pMax[dim]);
52 mid = std::partition(entityInfo.begin() + start, entityInfo.begin() + end, [&](const EntityInfo& pi) {
53 return pi.center[dim] < pmid;
54 }) - entityInfo.begin();
55 if (mid == start || mid == end) success = 0;
56 }
57
58 else if (nEntities <= 2 || splitMethod == SplitMethod::EqualCounts) {
59 mid = (start + end) >> 1;
60 std::nth_element(entityInfo.begin() + start, entityInfo.begin() + mid, entityInfo.begin() + end, [&](const EntityInfo& a, const EntityInfo& b) {
61 return a.center[dim] < b.center[dim];
62 });
63 success = 1;
64 }
65
66 else if (!success || splitMethod == SplitMethod::SAH) {
67 const int split = 30;
68 std::vector<std::pair<int, BoundingBox3f>> bucketInfos(split);
69 std::vector<BoundingBox3f> suffixBounds(split);
70 for (int i = start; i < end; i++) {
71 int bucket = split * (entityInfo[i].center[dim] - centerBounds.pMin[dim]) / maxD;
72 if (bucket == split) bucket--;
73 auto& [count, bounds] = bucketInfos[bucket];
74 count++;
75 bounds = BoundingBoxUnion(bounds, entityInfo[i].bounds);
76 }
77 for (int i = split - 1; i >= 0; i--) {
78 suffixBounds[i] = bucketInfos[i].second;
79 if (i + 1 < split) suffixBounds[i] = BoundingBoxUnion(suffixBounds[i], suffixBounds[i + 1]);
80 }
81 double minCost = DBL_MAX;
82 int count0 = 0, minCostBucket = -1;
83 BoundingBox3f prefix;
84 for (int i = 0; i + 1 < split; i++) {
85 const auto [count, bounds] = bucketInfos[i];
86 count0 += count;
87 prefix = BoundingBoxUnion(prefix, bounds);
88 double costI = 1 + (count0 * prefix.SurfaceArea() + (nEntities - count0) * suffixBounds[i + 1].SurfaceArea()) / totalBounds.SurfaceArea();
89 if (minCost > costI) {
90 minCost = costI;
91 minCostBucket = i;
92 }
93 }
94 if (minCost < nEntities) {
95 mid = std::partition(entityInfo.begin() + start, entityInfo.begin() + end, [&](const EntityInfo& pi) {
96 int bucket = split * (pi.center[dim] - centerBounds.pMin[dim]) / maxD;
97 if (bucket == split) bucket--;
98 return bucket <= minCostBucket;
99 }) - entityInfo.begin();
100 }
101 else {
102 //leaf node
103 mid = -1;
104 root->entityOffset = orderedEntites.size();
105 root->nEntites = nEntities;
106 for (int i = start; i < end; i++) {
107 orderedEntites.push_back(entites[entityInfo[i].EntityId]);
108 }
109 }
110 }
111 else assert(BVH_BUILD_ERROR);//should not be here!!!
112 if (mid != -1) {
113 assert(mid > start);
114 root->children[0] = RecursiveBuild(entityInfo, start, mid, nodeNumber, orderedEntites);
115 root->children[1] = RecursiveBuild(entityInfo, mid, end, nodeNumber, orderedEntites);
116 }
117 }
118 }
119 return root;
120}

158 if (linearBvhNodes.empty()) return std::nullopt;

162 std::vector<int> nextNodeStack;
163 int currentNode = 0, flag = 0;
164 std::optional<Intersection> result;
165 while (true) {
166 auto& node = linearBvhNodes[currentNode];
167 auto boundingIts = node.bounds.Intersection(R);
168 if (boundingIts.has_value()) {
169 if (node.nEntites > 0) {
170 for (int i = 0; i < node.nEntites; i++) {
171 auto its = entites[node.firstdEntityOffset + i]->intersect(R);
172 if (its.has_value()) {
173 flag = true;
174 double t = (its->position[0] - r.origin[0]) / r.direction[0];
175 if (R.timeMax > t) {
176 R.timeMax = t;
177 result = its;
178 }
179 }
180 }
181 if (nextNodeStack.empty()) break;
182 currentNode = nextNodeStack.back();
183 nextNodeStack.pop_back();
184 }
185 else {
186 if (isNegDir[node.splitAxis]) {
187 nextNodeStack.push_back(currentNode + 1);
188 currentNode = node.secondChildOrder;
189 }
190 else {
191 nextNodeStack.push_back(node.secondChildOrder);
192 currentNode = currentNode + 1;
193 }
194 }
195 }
196 else {
197 if (nextNodeStack.empty()) break;
198 currentNode = nextNodeStack.back();
199 nextNodeStack.pop_back();
